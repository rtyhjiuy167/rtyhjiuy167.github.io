#### 父组件向子组件传值

下面的写法，是在已经安装并配置了`unplugin-auto-import`插件的前提下写的。

写法一：

```vue
<!-- src/App.vue -->

<template>
  <List :msg="msg"></List>
</template>

<script setup>
// 使用 setup语言糖 父组件使用子组件
import List from "./components/list.vue";
let msg = ref("这是父组件传过去的数据");
</script>
```

```vue
<!-- src/components/list.vue -->

<template>
  <div>子组件 ==> {{ msg }}</div>
</template>

<script setup>
defineProps({
  msg: {
    type: String,
    default: "1111",
  },
});
</script>

<style>
</style>
```

写法二：

```vue
<!-- src/App.vue -->

<template>

<script setup>
// 使用 setup语言糖 父组件使用子组件
```

写法四：父传子，并修改其值

```vue
<!-- src/App.vue -->

<template>
  <List v-model:num="num"></List>
</template>

<script setup>
import List from './components/list.vue'
let num = ref(1);
</script>
```

```vue
<!-- src/components/list.vue -->

<template>
  <div>
    {{ num }}
    <button @click="btn">按钮</button>
  </div>
</template>

<script setup>
let prps = defineProps({
  num: {
    type: Number,
    default: 2,
  },
});
    // 固定写法 update: 变量名
const emit = defineEmits(["update:num"]);
const btn = () => {
  emit("update:num", 200);
};
</script >
```

#### 子组件向父组件传值

常用写法：

```vue
<!-- src/App.vue -->

<template>
  <List @fn="changeHome"></List>
</template>

<script setup>
// 使用 setup语言糖 父组件使用子组件
import List from "./components/list.vue";
let changeHome = (n) => {
  console.log(n.value);
};
</script>
```

```vue
<!-- src/components/list.vue -->

<template>
  <div>
    子组件 ==> {{ num }}
    <button @click="changeNum">按钮</button>
  </div>
</template>

<script setup>
let num =ref(100)
const emit = defineEmits(["fn"]);
const changeNum = ()=>{
  emit('fn',num)
}
</script >

<style>
</style>
```

其他写法：

```vue
<!-- src/App.vue -->

<template>
  <List @fn="changeHome"></List>
</template>

<script >
// 使用 setup语言糖 父组件使用子组件
import List from "./components/list.vue";
export default {
  components: { List },
  setup() {
    let changeHome = (n) => {
      console.log(n.value);
    };
    return {
      changeHome,
    };
  },
};
</script>
```

```vue
<!-- src/components/list.vue -->

<template>
  <div>
    子组件 ==> {{ num }}
    <button @click="changeNum">按钮</button>
  </div>
</template>

<script >
export default {
  props: ["msg"],
  setup(props, { emit }) {
    let num = ref(100);
    const changeNum = () => {
      emit("fn", num);
    };
    return {
      num,
      changeNum,
    };
  },
};
</script>

<style>
</style>
```

#### 任意组件间传值

使用 Provide 与 Inject，且数据可以更改

```vue
<!-- src/App.vue -->

<template>
  <A></A>
</template>

<script setup>
import A from "./components/A.vue";
let num = ref(100);
provide("num", num);
</script>

```

```vue
<template>{{num }}</template>

<script setup>
const num = inject("num");
</script>
```
