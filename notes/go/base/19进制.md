```go
package main

import "fmt"

//进制
// 二进制：0,1，满2进1
// 十进制：0~9，满10进1
// 八进制：0~7 满8进1，以数字0开头
/*
十六进制：0~9及A~F，满16进1，以0x或0X开头表示，且A~F不区分大小写
	A表示10，B表示11，C表示12，D表示13，E表示14，F表示15
*/
func main() {

	//八进制
	var j int = 021      // 8^0*1 + 8^1*2 = 17
	fmt.Println("j=", j) //输出为十进制 17

	//十六进制
	var k int = 0x34A    // 16^0*10 + 16^1*4 + 16^2*3 = 842
	fmt.Println("k=", k) //输出为十进制 842

	//十进制转二进制
	//将该数不断除以2，直到商为0，然后将每步得到的余数倒过来，就是对应的二进制
	/*
	 56 / 2 = 28···0
	 28 / 2 = 14···0
	 14 / 2 = 7 ···0
	  7 / 2 = 3 ···1
	  3 / 2 = 1 ···1
	  1 / 2 = 0 ···1
	  结果即为 11000
	*/
	var i int = 56
	//二进制输出  %b
	fmt.Printf("%b\n", i) //111000

	//十进制转八进制
	//将该数不断除以8，直到商为0，然后将每步得到的余数倒过来，就是对应的八进制
	i = 156
	/*
		 156 / 8 = 19···4
		  19 / 8 = 2 ···3
		   2 / 8 = 0 ···2
		结果即为 234
	*/
	//八进制输出  %o
	fmt.Printf("%o\n", i) //234

	//十进制转十六进制
	//将该数不断除以16，直到商为0，然后将每步得到的余数倒过来，就是对应的十六进制
	i = 3564
	/*
		 3564 / 16 = 222···12
		  222 / 16 = 13 ···14
		   13 / 16 = 0 ···13
		结果即为 13 14 12 --> dec
	*/
	//八进制输出  %x
	fmt.Printf("%x\n", i) //dec
}
```

<h4>二进制转八进制与十六进制</h4>

```
二进制转成八进制
将二进制每三位一组（从第一位（个位？！）开始组合），转成对应的八进制数即可
例如：11010101
    11                 010                    101
2^0*1+2^1*1      2^0*0+2^1*1+2^2*0     2^0*1+2^1*0+2^2*1
     3                  2                      5
结果即为 0325

二进制转成十六进制
将二进制每四位一组（从第一位（个位？！）开始组合），转成对应的十六进制数即可
例如：11010101
        1101                         0101
2^0*1+2^1*0+2^2*1+2^3*1     2^0*1+2^1*0+2^2*1+2^3*0
	     13                           5
          D                           5
结果即为 0xD5
```

<h4>八进制与十六进制转二进制</h4>

```
八进制转二进制
将八进制数的每一位，转成对应的一个3位的二进制数即可，最后一个能用一位就用一位，不行就二位，再不行就三位
例如： 0237  //0是八进制开头用的
 2   3   7
10  011 111
结果即为 10011111

十六进制转二进制
将十六进制数的每一位，转成对应的一个4位的二进制数即可，最后一个能用一位就用一位，不行就二位，再不行就三位，还不行就四位
例如： 0x237  //0x是十六进制开头用的
 2   3    7
10  0011 0111
结果即为 1000110111
```
